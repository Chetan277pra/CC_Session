#include <bits/stdc++.h>
using namespace std;
using ll = long long;
 
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int t;
    if (!(cin >> t)) return 0;
    while (t--) {
        int n, k;
        cin >> n >> k;
        vector<ll> vec1(n), vec2(n);
        for (int i = 0; i < n; ++i) cin >> vec1[i];
        for (int i = 0; i < n; ++i) cin >> vec2[i];
        for (int i = 0; i < n; ++i) if (vec1[i] > vec2[i]) swap(vec1[i], vec2[i]);
        ll base = 0;
        vector<pair<ll,ll>> seg(n);
        for (int i = 0; i < n; ++i) {
            base += (vec2[i] - vec1[i]);
            seg[i] = {vec1[i], vec2[i]};
        }
        sort(seg.begin(), seg.end());
        bool intersect = false;
        ll maxR = LLONG_MIN;
        for (int i = 0; i < n; ++i) {
            if (seg[i].first <= maxR) { intersect = true; break; }
            maxR = max(maxR, seg[i].second);
        }
        if (intersect) {
            cout << base << '\n';
            continue;
        }
        ll min_gap = LLONG_MAX;
        for (int i = 1; i < n; ++i) min_gap = min(min_gap, seg[i].first - seg[i-1].second);
        cout << base + 2 * min_gap << '\n';
    }
    return 0;
}
